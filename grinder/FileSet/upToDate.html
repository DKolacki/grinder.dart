bool upToDate(FileSet other)

Returns whether this file set exists, and is not older then the given FileSet.

Source 

bool upToDate(FileSet other) {
  if (!exists) {
    return false;
  } else {
    return !other.lastModified.isAfter(lastModified);
  }
}
