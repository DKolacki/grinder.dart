DateTime lastModified

Implementation 

DateTime get lastModified {
  DateTime time = new DateTime.fromMillisecondsSinceEpoch(0);

  files.forEach((f) {
    DateTime modified = f.lastModifiedSync();
    if (modified.isAfter(time)) {
      time = modified;
    }
  });

  return time;
}
